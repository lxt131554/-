### Task 4: 预警提醒

**warnings.js:**
```js
export function getWarnings(stages) {
  const now = new Date()
  return {
    overdue: stages.filter(s => s.planEnd && new Date(s.planEnd) < now && s.status !== 'completed'),
    nearDeadline: stages.filter(s => {
      if (!s.planEnd || s.status === 'completed') return false
      const daysLeft = Math.ceil((new Date(s.planEnd) - now) / (1000*60*60*24))
      return daysLeft <= 2 && daysLeft >= 0
    })
  }
}
```
